% ---------- main.tex ----------
%
% Main entry point for the resume document.
% Defines document class, loads packages, configures formatting, and assembles sections.
%
% Build: pdflatex main.tex (or lualatex/xelatex for Unicode fonts)

% ---- document class ----
\documentclass[11pt]{article}       % set main text size

% ---- core packages ----
\usepackage{import}                 % modular file structure
\usepackage{xparse}                 % advanced command definitions
\usepackage{iftex}                  % engine detection for conditional loading

% ---- page layout ----
\usepackage[letterpaper,            % paper size

            left=0.5in,             % left page margin

\pagestyle{empty}                   % disable page numbering
\raggedright                        % disable text justification

% ---- typography and formatting ----
\usepackage{microtype}              % micro-typographic enhancements
\usepackage{enumitem}               % list customization (bullet points)
\usepackage{titlesec}               % section title customization
\usepackage[hidelinks]{hyperref}    % hyperlinks without visible borders

% Ensure PDF output is Unicode-compliant and machine-readable (ATS-friendly)

\fi

% ---- load commands ----
